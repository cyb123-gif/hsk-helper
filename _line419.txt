    function renderTrackList(){ const items=[]; for(let [pat,entry] of usedMap.entries()){ const c=getConfirmedCount(entry); if(c>0) items.push({pattern:pat, level:entry.level, confirmed:c, total:entry.count}); } items.sort((a,b)=>a.level-b.level); if(items.length===0){ trackListDiv.innerHTML='<div class="empty-detail">暂无已确认的正确句子</div>'; return; } let html=''; for(let item of items){ const acquired=item.confirmed>=ACQUIRED_THRESHOLD; const entry=usedMap.get(item.pattern); let latestDate=''; if(entry&&entry.examples.length){ const dates=entry.examples.map(e=>e.date).filter(Boolean); if(dates.length) latestDate=dates.sort().reverse()[0]; } html+=`<div class="track-item${acquired?' acquired':''}" data-pattern="${escapeHtml(item.pattern)}"><span>HSK${item.level} ${escapeHtml(item.pattern.slice(0,30))}${latestDate?' <small style="color:#94a3b8;">'+formatDate(latestDate)+'</small>':''}</span><span>✅${item.confirmed}${acquired?' <span class=\"acquired-badge\">已习得</span>':''}</span></div>`; } trackListDiv.innerHTML=html; document.querySelectorAll('.track-item').forEach(el=>el.addEventListener('click',()=>{ const pat=el.dataset.pattern; const entry=usedMap.get(pat); if(entry) openGrammarDetailModalWithEntry(pat,entry); })); }
